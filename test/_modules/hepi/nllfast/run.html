[docs]	def _check_input(self,p: Input)-> bool:
		"""Checks input parameter for compatibility with Prospino"""
		if p.mu_f != 1. or  p.mu_r != 1.:
			warnings.warn("NLL-fast does not support varying the scales manually.")
			return False
		if p.pdf_lo != "cteq6l1"  or  p.pdf_nlo != "cteq66":
			warnings.warn("NLL-fast does not support all pdfs (CTEQ6L1 and CTEQ66 allowed defaults).")
			return False
		return True

		rp = super()._prepare(p,**kwargs)

			open(rp.execute, "w").write("#!/bin/sh\n"+
				"{path}/nll-fast {proc} {pdf} {sq} {gl}> {out}".format(
				path=self.get_path(),out=rp.out_file , proc = d["nf_final_state_in"], sq = d["nf_squark_mass"], gl = d["nf_gluino_mass"], pdf = d["nf_pdf"]))
